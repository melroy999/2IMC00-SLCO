int turn[5] = {0,0,0,0,0};
int b[5] = {0,0,0,0,0};
int c[5] = {0,0,0,0,0};

proctype p1 {	
int curr = 0;
do
:: true ->
	if
	:: curr >= 0 && curr < 5 ->
		store(turn[curr], 0);
		store(b[curr], 1);
		if
		:: load(turn[curr], 0) ->
			skip;
		:: load(turn[curr], 1) || load(turn[curr], 2) ->
			c[curr] = 1;
			b[curr] = 0;
			break;
		fi;
	fi;
od;
}
